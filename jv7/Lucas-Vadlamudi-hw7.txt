final class Node<E> implements Iterable<E> {
  // private fields
  private E val;
  private Node<E> next;

  // (1 point) implement constructor
  public Node(E val, Node<E> node) {
    this.val = val;
    this.next = node;
  }

  // (2 points) iterator() returns a NodeIterator object for this object
  // pass this so the iterator starts at the current node
  @Override
  public NodeIterator<E> iterator() {
    return new NodeIterator<E>(this);
  }

  // (2 points) getter and setter methods for the private fields
  public E getVal() {
    return val;
  }

  public void setVal(E v) {
    val = v;
  }

  public Node<E> getNext() {
    return next;
  }

  public void setNext(Node<E> node) {
    next = node;
  }

  // *** NodeIterator as an inner class: 10 points
  // Having NodeIterator as an inner class of Node makes sense
  // correct class header as given in the problem statement
  class NodeIterator<E> implements Iterator<E> {
    private Node<E> p; // given

    // (1 points) constructor
    public NodeIterator(Node<E> n) {
      p = n;
    }

    // (9 points) methods to implement the Iterator interface
    //// (2 points) hasNext()
    // p becomes null when we go past the last node so use that to check
    @Override
    public boolean hasNext() {
      return p != null;
    }

    //// (7 points) next()
    @Override
    public E next() {
      if (hasNext() == false) // if there is no next value, end
        throw new NoSuchElementException();
      // save the value first before moving p or else we lose it
      E elem = p.getVal();
      p = p.getNext();
      return elem;
    }
  } // end of NodeIterator

  // Total 10 points for the two methods: mysum, print

  //// 7 points
  // ? extends Number, we can pass Node<Double> or Node<Float>/ Node<Integer>
  // without wildcard java disallows Node<Double> where Node<Number> expected
  static double mysum(Iterable<? extends Number> n) {
    double sum = 0.0;
    for (Number num : n) {
      sum += num.doubleValue();
    }
    return sum;
  }

  //// 3 points
  // T can be anything, loop and print each element with a space
  static <T> void print(Iterable<T> n) {
    for (T elem : n) {
      System.out.print(elem + " ");
    }
    System.out.println();
  }

  // Required: (20 points) Expand the main method (see the pdf for

// Total 55 points for the MyQueue class

class MyQueue<E> implements Iterable<E>, Cloneable, Comparable<MyQueue<E>> {
  protected Node<E> head;
  protected Node<E> tail;
  protected int length;

  @Override
  public Iterator<E> iterator() {
    return head.iterator();
  }

  // helper method reverse() *** to be used in clone()
  public MyQueue<E> reverse() {
    Node<E> l = null;
    for (E e : head)
      l = new Node<E>(e, l);
    return (new MyQueue<E>(l));
  }

  // Task 1: override Object.clone() (5 points)
  // make empty queue then add each element in order with for each
  @Override
  public MyQueue<E> clone() {
    MyQueue<E> copy = new MyQueue<E>();
    if (head == null)
      return copy;
    for (E e : this) {
      copy.add(e);
    }
    return copy;
  }

  @Override
  public int compareTo(MyQueue<E> list) {
    if (this.length < list.length)
      return -1;
    if (this.length == list.length)
      return 0;
    return 1;
  }

  // Task 2: override Object.equals() (5 points)
  @Override
  public boolean equals(Object o) {
    // check if its even a MyQueue first
    if (!(o instanceof MyQueue))
      return false;
    MyQueue<?> other = (MyQueue<?>) o;
    if (this.length != other.length)
      return false;
    if (this.length == 0)
      return true;
    // walk both iterators side by side and compare each pair
    Iterator<E> it1 = this.iterator();
    Iterator<?> it2 = other.iterator();
    while (it1.hasNext()) {
      E e1 = it1.next();
      Object e2 = it2.next();
      if (e1.equals(e2) == false)
        return false;
    }
    return true;
  }

  @Override
  public int hashCode() {
    return length;
  }

  // two constructors (implementing the one-arg constructor 5 points)
  // no-arg constructor given
  public MyQueue() {
    head = null;
    tail = null;
    length = 0;
  }

  // Task 3: one-arg constructor (10 points)
  // start as empty then call add for each element to keep same order
  public MyQueue(Iterable<E> iterable) {
    head = null;
    tail = null;
    length = 0;
    for (E e : iterable) {
      add(e);
    }
  }

  // Task 4: total 15 points for toString(), add() and remove()
  // toString() (5 points)
  @Override
  public String toString() {
    if (head == null)
      return "(empty)";
    String result = "";
    Node<E> curr = head;
    while (curr != null) {
      if (curr == head && curr == tail) {
        result = result + "(head: " + curr.getVal() + ")";
      } else if (curr == head) {
        result = result + "(head: " + curr.getVal() + ")";
      } else if (curr == tail) {
        result = result + "-> (tail: " + curr.getVal() + ")";
      } else {
        result = result + "-> (" + curr.getVal() + ")";
      }
      curr = curr.getNext();
    }
    return result;
  }

  // add() (5 points)
  // if queue is empty both head and tail point to new node
  // otherwise attach new node after tail and update tail pointer
  public void add(E item) {
    Node<E> newNode = new Node<E>(item, null);
    if (head == null) {
      head = newNode;
      tail = newNode;
    } else {
      tail.setNext(newNode);
      tail = newNode;
    }
    length = length + 1;
  }

  // remove() (5 points)
  // save the head value, move head forward, if now empty set tail to null too
  public E remove() {
    if (head == null)
      throw new NoSuchElementException();
    E val = head.getVal();
    head = head.getNext();
    if (head == null) {
      tail = null;
    }
    length = length - 1;
    return val;
  }

  // given
  public E peek() {
    return head.getVal();
  }

  // given
  public int getLength() {
    return length;
  }
