subset :: Eq a => Set a -> Set a -> Bool
subset [] b = True -- base case
subset (a:as) b = if isElem a b == True then subset as b else False -- checks if every element in the first set appears in the second set recursively
-- subset checks whether every element in the first set appears in the second set
-- the base case is when the first set is empty, which always returns True
-- because the empty set is a subset of any set
-- in the recursive case, we check if the current element a is in b using isElem
-- if it is, we continue checking the rest of the elements
-- if it is not, we immediately return False

scalarproduct :: [Int] -> [Int] -> Int
scalarproduct as bs = additionHelper [a*b | (a,b) <- zip as bs] -- multiplies matching elements from two lists (via list comp) and adds them up

additionHelper :: [Int] -> Int
additionHelper [] = 0 -- base case
additionHelper (x:xs) = x + additionHelper xs -- helper function adds up all numbers in a list using recursion
